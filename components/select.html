自定义内容 #

代码示例

<template>
    <pied-select v-model="inputValue" filterable data="data">
        <template v-slot="scope">
            <div class="flex"><p>{{scope.data.value}}</p><p>{{scope.data.label}}</p></div>
        </template>
    </pied-select>
</template>

<script setup>
import { ref } from 'vue'
const inputValue = ref('')

    }
])
</script>
